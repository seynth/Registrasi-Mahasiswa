```cpp
void tambahMahasiswa(Mahasiswa*& head, string nim, string nama, string jurusan) {
    Mahasiswa* baru = new Mahasiswa;  // Alokasi memori untuk node baru
    baru->nim = nim;                   // Isi atribut NIM
    baru->nama = nama;                 // Isi atribut nama
    baru->jurusan = jurusan;           // Isi atribut jurusan
    baru->relasiHead = nullptr;        // Inisialisasi list relasi kosong
    baru->next = nullptr;              // Set pointer next ke null
    
    if (head == nullptr) {             // Jika list kosong
        head = baru;                   // Node baru menjadi head
    } else {                           // Jika list tidak kosong
        Mahasiswa* temp = head;
        while (temp->next != nullptr) { // Traverse ke node terakhir
            temp = temp->next;
        }
        temp->next = baru;             // Tambah node baru di akhir
    }
    // INSERT LAST: Menambahkan node di akhir list
}
```

### 3. Fungsi `cariMahasiswa` - Search Parent

```cpp
Mahasiswa* cariMahasiswa(Mahasiswa* head, string nim) {
    Mahasiswa* temp = head;
    while (temp != nullptr) {          // Traverse seluruh list
        if (temp->nim == nim) {        // Jika NIM cocok
            return temp;               // Kembalikan pointer ke node
        }
        temp = temp->next;             // Pindah ke node berikutnya
    }
    return nullptr;                    // Tidak ditemukan
}
```

### 4. Fungsi `hapusMahasiswa` - Delete Parent dengan Relasi (Menu 4)
```cpp
void hapusMahasiswa(Mahasiswa*& head, string nim) {
    if (head == nullptr) return;      // Validasi list tidak kosong
    
    Mahasiswa* hapus = nullptr;
    
    // Kasus 1: node yang dihapus adalah head
    if (head->nim == nim) {
        hapus = head;
        head = head->next;           // Update head
    } else { // Kasus 2: node di tengah/akhir
        Mahasiswa* temp = head;
        while (temp->next != nullptr && temp->next->nim != nim) {
            temp = temp->next;       // Cari node sebelum yang akan dihapus
        }
        if (temp->next == nullptr) return; // Tidak ditemukan
        hapus = temp->next;          // Simpan node yang akan dihapus
        temp->next = hapus->next;    // By-pass node yang dihapus
    }
    
    // Hapus semua relasi child terlebih dahulu
    Relasi* relTemp = hapus->relasiHead;
    while (relTemp != nullptr) {
        Relasi* hapusRel = relTemp;
        relTemp = relTemp->next;
        delete hapusRel;             // Dealokasi memori relasi
    }
    
    delete hapus;                    // Dealokasi memori mahasiswa
}
// DELETE FIRST: Berpotensi menghapus dari depan jika head yang dihapus
```

### 6. Fungsi `hapusMataKuliah` - Delete Child dengan Cleanup Relasi (Menu 5)
```cpp
void hapusMataKuliah(Mahasiswa*& headMahasiswa, MataKuliah*& headMataKuliah, string kode) {
    // Hapus node mata kuliah dari list
    // ... (logika penghapusan mirip dengan hapusMahasiswa)
    
    // Cleanup relasi di semua mahasiswa
    Mahasiswa* mhsTemp = headMahasiswa;
    while (mhsTemp != nullptr) {
        Relasi* rel = mhsTemp->relasiHead;
        Relasi* prev = nullptr;
        
        while (rel != nullptr) {
            if (rel->mataKuliah->kode == kode) {
                // Hapus relasi dari list mahasiswa
                if (prev == nullptr) {
                    mhsTemp->relasiHead = rel->next;  // Hapus dari depan
                } else {
                    prev->next = rel->next;           // Hapus dari tengah/akhir
                }
                Relasi* hapusRel = rel;
                rel = rel->next;
                delete hapusRel;
            } else {
                prev = rel;
                rel = rel->next;
            }
        }
        mhsTemp = mhsTemp->next;
    }
}
// DELETE CHILD PADA PARENT TERTENTU: Menghapus relasi di semua mahasiswa
```

### 7. Fungsi `tambahRelasi` - Hubungkan Parent ke Child (Menu 3)
```cpp
void tambahRelasi(Mahasiswa* headMahasiswa, MataKuliah* headMataKuliah, 
                  string nim, string kodeMataKuliah, string semester) {
    
    Mahasiswa* mhs = cariMahasiswa(headMahasiswa, nim);      // Cari parent
    MataKuliah* mk = cariMataKuliah(headMataKuliah, kodeMataKuliah); // Cari child
    
    // Validasi data ditemukan
    if (mhs == nullptr || mk == nullptr) return;
    
    // Cek duplikasi relasi
    Relasi* relTemp = mhs->relasiHead;
    while (relTemp != nullptr) {
        if (relTemp->mataKuliah->kode == kodeMataKuliah) return;
        relTemp = relTemp->next;
    }
    
    // Tambah relasi baru
    Relasi* baru = new Relasi;
    baru->mataKuliah = mk;            // Pointer ke child
    baru->semester = semester;
    baru->next = nullptr;
    
    if (mhs->relasiHead == nullptr) {
        mhs->relasiHead = baru;       // INSERT FIRST untuk relasi
    } else {
        Relasi* temp = mhs->relasiHead;
        while (temp->next != nullptr) {
            temp = temp->next;
        }
        temp->next = baru;            // INSERT LAST untuk relasi
    }
    
    mk->jumlahPeminat++;              // Update counter peminat
}
```

```cpp
void tampilkanSemuaMahasiswa(Mahasiswa* head) {
    Mahasiswa* temp = head;
    while (temp != nullptr) {
        int jumlahMK = 0;
        Relasi* relTemp = temp->relasiHead;
        
        // Hitung jumlah child (mata kuliah)
        while (relTemp != nullptr) {
            jumlahMK++;
            relTemp = relTemp->next;
        }
        
        // Tampilkan data parent dengan jumlah child
        cout << setw(15) << left << temp->nim
             << setw(25) << left << temp->nama
             << setw(20) << left << temp->jurusan
             << setw(20) << left << jumlahMK << endl;
        
        temp = temp->next;
    }
}
// HITUNG JUMLAH CHILD: Menghitung mata kuliah per mahasiswa
```

```cpp
        cout << setw(10) << left << mk->kode
             << setw(30) << left << mk->nama
             << setw(8) << left << mk->sks
             << setw(10) << left << mk->tingkat
             << setw(12) << left << relTemp->semester << endl;

    }
}
```

void tampilkanStatistikPeminat(MataKuliah* head) {
    MataKuliah* terbanyak = head;
    MataKuliah* tersedikit = head;
    
    while (temp != nullptr) {
        // Cari mata kuliah dengan peminat terbanyak
        if (temp->jumlahPeminat > terbanyak->jumlahPeminat) {
            terbanyak = temp;
        }
        // Cari mata kuliah dengan peminat tersedikit
        if (temp->jumlahPeminat < tersedikit->jumlahPeminat) {
            tersedikit = temp;
        }
        temp = temp->next;
    }
}
